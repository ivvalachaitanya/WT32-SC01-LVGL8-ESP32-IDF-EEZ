üöÄ Event Callbacks (actions.c)

Example implementation:

#include "actions.h"
#include "lvgl.h"
#include "esp_log.h"

static const char *TAG = "UI_ACTIONS";

void action_press(lv_event_t *e) {
    int param = (int)lv_event_get_user_data(e);
    ESP_LOGI(TAG, "[B%d] Pressed, parameter = %d", param, param);
}

void action_pressing(lv_event_t *e) {
    int param = (int)lv_event_get_user_data(e);
    ESP_LOGI(TAG, "[B%d] Pressing‚Ä¶", param);
}

void action_shclick(lv_event_t *e) {
    int param = (int)lv_event_get_user_data(e);
    ESP_LOGI(TAG, "[B%d] Short Click", param);
}

void action_lngclick(lv_event_t *e) {
    int param = (int)lv_event_get_user_data(e);
    ESP_LOGI(TAG, "[B%d] Long Press", param);
}

void action_click(lv_event_t *e) {
    int param = (int)lv_event_get_user_data(e);
    ESP_LOGI(TAG, "[B%d] Click", param);
}

void action_release(lv_event_t *e) {
    int param = (int)lv_event_get_user_data(e);
    ESP_LOGI(TAG, "[B%d] Release", param);
}
